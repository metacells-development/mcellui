```typescript
function TaskItemPreview() {
  const { spacing } = useTheme();
  const [tasks, setTasks] = useState([
    { id: '1', title: 'Complete project review', completed: false, priority: 'high' as const, dueDate: '2024-01-20' },
    { id: '2', title: 'Send weekly report', completed: true },
  ]);

  const toggleTask = (id: string) => {
    setTasks(t => t.map(task => task.id === id ? { ...task, completed: !task.completed } : task));
  };

  return (
    <View style={{ gap: spacing[2] }}>
      {tasks.map(task => (
        <TaskItem
          key={task.id}
          task={task}
          onToggle={() => toggleTask(task.id)}
          onPress={() => Alert.alert('Task Details')}
          onDelete={() => Alert.alert('Delete')}
        />
      ))}
    </View>
  );
}
```
